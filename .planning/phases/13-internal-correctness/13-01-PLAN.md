// Compares IParameterSymbol by Type.Name and param Name -- does NOT use full namespace
internal sealed class FluentParameterComparer : IEqualityComparer<IParameterSymbol>
{
    public bool Equals(IParameterSymbol? x, IParameterSymbol? y)
    {
        return x.Type.Name == y.Type.Name && x.Name == y.Name;
    }
    public int GetHashCode(IParameterSymbol? obj)
    {
        return (obj.Type.Name.GetHashCode() * 397) ^ obj.Name.GetHashCode();
    }
}
```

// Equals: same FluentType AND overlapping Names
// GetHashCode: delegates to FluentType.GetHashCode() only (ignores Names)
internal class FluentMethodParameter : IEquatable<FluentMethodParameter>
{
    public bool Equals(FluentMethodParameter? other)
    {
        if (!FluentType.Equals(other.FluentType)) return false;
        return Names.Overlaps(other.Names);
    }
    public override int GetHashCode() => FluentType.GetHashCode();
}
```
